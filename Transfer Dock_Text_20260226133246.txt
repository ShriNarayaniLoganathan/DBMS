
create table bank_acc(accno varchar2(20) primary key,name varchar2(20), deposit number);
set serveroutput on
declare
vaccno varchar2(20) := '&vaccno';
vname varchar2(20):= '&vname';
vdeposit number := &vdeposit;
zerodeposit exception;
begin
if vdeposit <= 0 then
raise zerodeposit;
else
insert into bank_acc(accno,name, deposit) values (vaccno,vname, vdeposit);
dbms_output.put_line('Record successfully inserted');
end if;
exception
when zerodeposit then
dbms_output.put_line('Invalid amount');
end;
/
